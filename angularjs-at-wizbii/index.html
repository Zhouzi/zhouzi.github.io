Principes & Objectifs

AngularJS n'est utilisé que pour améliorer "l'expérience utilisateur" mais pas en tant qu'outil central.

  • Utilisation au minimum
  • Simple, réutilisable

Comment ?

  • Directives
  • Création de "classes" plutôt que controlleur
  • Séparation et indépendance

app.controller("PublicationsController", [
  "$scope",
  "PublicationModel",
  function ($scope, PublicationModel) {
    var sth = null;

    $scope.publications = [];

    (function (publications) {
      angular.forEach(publications, function (publication) {
        $scope.publications.push(new PublicationModel(publication));
      });
    })(window.data.publications);
  }
]);



app.service("PublicationModel", [
    "PublicationService",
    function (PublicationService) {
        function PublicationModel (attrs) {
            angular.extend(this, attrs);
        }

        PublicationModel.prototype.like = function () {
            PublicationService.like(this.id);
        };

        return PublicationModel;
    }
]);
